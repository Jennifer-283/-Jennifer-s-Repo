import math


def solve_quadeq(a: float, b: float, c: float):
    """Robust solver that avoids cancellation in (-b ± sqrt(D))."""
    D = b*b - 4*a*c
    if D < 0:
        raise ValueError("Discriminant must be non-negative for real roots")
    sqrtD = math.sqrt(D)

    if b >= 0:
        x1 = (-b - sqrtD) / (2*a)
        x2 = (-2*c) / (b + sqrtD)
    else:
        x1 = (-2*c) / (b - sqrtD)
        x2 = (-b + sqrtD) / (2*a)

    return x1, x2


def solve_naive(a, b, c):
    D = b*b - 4*a*c
    sqrtD = math.sqrt(D)
    x1 = (-b - sqrtD) / (2*a)
    x2 = (-b + sqrtD) / (2*a)
    return x1, x2


def solve_robust(a, b, c):
    return solve_quadeq(a, b, c)


def residual(a, b, c, x):
    return a*x*x + b*x + c


if __name__ == "__main__":
    cases = [
        (1.0,  1e15, 1e14),
        (1.0, -1e15, 1e14),
    ]

    for a, b, c in cases:
        print(f"\nCase: a={a}, b={b}, c={c}")
        x_naive = solve_naive(a, b, c)
        x_robust = solve_robust(a, b, c)

        print("Method    | x1                     | x2                     | res1                 | res2")
        print("----------|------------------------|------------------------|----------------------|----------------------")
        for name, (x1, x2) in [("naive", x_naive), ("robust", x_robust)]:
            r1 = residual(a, b, c, x1)
            r2 = residual(a, b, c, x2)
            print(f"{name:<9}| {x1:22.15g} | {x2:22.15g} | {r1:20.13g} | {r2:20.13g}")
